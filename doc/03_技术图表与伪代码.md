## 3 关键算法伪代码

### 3.1 行为树 Tick 算法

```
Algorithm: BehaviorTreeTick
Input: root_node, context
Output: NodeStatus

function tick_tree(root_node, context):
    context.tick_count += 1
    context.elapsed_time = current_time - start_time
    
    status = root_node.tick(context)
    
    if status != RUNNING:
        engine._running = false
        notify_status_change("completed", status)
    
    return status
```

### 3.2 Sequence 节点 Tick 算法

```
Algorithm: SequenceTick
Input: children[], current_index, context
Output: NodeStatus

function sequence_tick(children, current_index, context):
    if children is empty:
        return SUCCESS
    
    while current_index < len(children):
        child = children[current_index]
        
        // 跳过禁用节点
        if not child.config.enabled:
            current_index += 1
            continue
        
        status = child.tick(context)
        
        switch status:
            case RUNNING:
                return RUNNING    // 暂停，等待下次tick
            case FAILURE:
                current_index = 0 // 重置索引
                return FAILURE    // 短路返回
            case SUCCESS:
                current_index += 1  // 继续下一个
        
    // 所有子节点成功
    current_index = 0
    return SUCCESS
```

### 3.3 Parallel 节点 Tick 算法

```
Algorithm: ParallelTick
Input: children[], cached_statuses, success_policy, context
Output: NodeStatus

function parallel_tick(children, cached_statuses, success_policy, context):
    if children is empty:
        return SUCCESS
    
    success_count = 0
    failure_count = 0
    running_children = []
    enabled_count = 0
    
    for i, child in enumerate(children):
        if not child.config.enabled:
            continue
        enabled_count += 1
        
        // 跳过已完成的子节点
        if i in cached_statuses:
            status = cached_statuses[i]
            if status == SUCCESS: success_count += 1; continue
            if status == FAILURE: failure_count += 1; continue
        
        // Tick 未完成的子节点
        status = child.tick(context)
        
        switch status:
            case SUCCESS:
                cached_statuses[i] = SUCCESS
                success_count += 1
            case FAILURE:
                cached_statuses[i] = FAILURE
                failure_count += 1
            case RUNNING:
                running_children.append(child)
    
    // require_one 策略：任一成功即返回
    if success_policy == REQUIRE_ONE and success_count > 0:
        abort_running(running_children)
        return SUCCESS
    
    // 仍有运行中的子节点
    if running_children is not empty:
        return RUNNING
    
    // 所有子节点执行完毕
    if success_policy == REQUIRE_ALL:
        return SUCCESS if success_count == enabled_count else FAILURE
    else:
        return SUCCESS if success_count > 0 else FAILURE
```

### 3.4 撤销/重做算法

```
Algorithm: CommandManager
State: undo_stack, redo_stack, max_history=100, is_executing=false

function execute(command):
    if is_executing: return false
    is_executing = true
    try:
        if command.execute():
            undo_stack.push(command)
            redo_stack.clear()
            if len(undo_stack) > max_history:
                undo_stack.pop_oldest()
            return true
        return false
    finally:
        is_executing = false

function undo():
    if undo_stack is empty: return false
    is_executing = true
    try:
        command = undo_stack.pop()
        if command.undo():
            redo_stack.push(command)
            return true
        undo_stack.push(command)  // 恢复失败，放回
        return false
    finally:
        is_executing = false

function redo():
    if redo_stack is empty: return false
    is_executing = true
    try:
        command = redo_stack.pop()
        if command.redo():
            undo_stack.push(command)
            return true
        redo_stack.push(command)  // 重做失败，放回
        return false
    finally:
        is_executing = false
```

### 3.5 序列化算法

```
Algorithm: Serialize
Input: root_node
Output: Dict (JSON-serializable)

function serialize(root_node):
    nodes = {}
    connections = []
    
    // 递归收集所有节点和连线
    function collect_nodes(node):
        nodes[node.node_id] = node.to_dict()
        for child in node.children:
            connections.append({
                parent_id: node.node_id,
                child_id: child.node_id
            })
            collect_nodes(child)
    
    if root_node:
        collect_nodes(root_node)
    
    return {
        version: "2.0",
        format_type: "behavior_tree_standalone",
        metadata: { created_at, modified_at, app_version },
        root_node: root_node.node_id,
        nodes: nodes,
        connections: connections
    }
```

### 3.6 反序列化算法

```
Algorithm: Deserialize
Input: data (Dict)
Output: Tuple (root_node, canvas_state, editor_state)

function deserialize(data):
    nodes_data = data["nodes"]
    connections = data["connections"]
    root_id = data["root_node"]
    
    // 提取画布和编辑器状态
    canvas_state = data.get("canvas_state", {})
    editor_state = data.get("editor_state", {})
    
    if not root_id or not nodes_data:
        return (null, canvas_state, editor_state)
    
    // 第一遍：创建所有节点实例
    nodes = {}
    for node_id, node_data in nodes_data:
        node_class = NodeRegistry.get(node_data["type"])
        if node_class:
            nodes[node_id] = node_class.from_dict(node_data)
    
    // 第二遍：重建连接关系
    for conn in connections:
        parent = nodes[conn["parent_id"]]
        child = nodes[conn["child_id"]]
        if parent and child:
            parent.add_child(child)
    
    return (nodes[root_id], canvas_state, editor_state)
```

### 3.7 自动布局算法

```
Algorithm: AutoLayout
Input: nodes_data, root_id
Output: positions Dict[node_id → (x, y)]

const NODE_WIDTH = 180
const NODE_HEIGHT = 80
const H_GAP = 50
const V_GAP = 100

function auto_layout(nodes_data, root_id):
    positions = {}
    
    function get_children(node_id):
        return nodes_data[node_id]["children"]
    
    function calc_subtree_width(node_id):
        children = get_children(node_id)
        if children is empty:
            return NODE_WIDTH
        total = sum(calc_subtree_width(c) for c in children)
        return max(total, NODE_WIDTH)
    
    function layout_node(node_id, x, y):
        positions[node_id] = (x, y)
        children = get_children(node_id)
        if children is not empty:
            total_width = sum(calc_subtree_width(c) for c in children)
            current_x = x - total_width / 2
            for child in children:
                child_width = calc_subtree_width(child)
                layout_node(child, current_x + child_width/2, y + NODE_HEIGHT + V_GAP)
                current_x += child_width
    
    layout_node(root_id, 400, 100)
    return positions
```

```
function get_dd_code(key_name):
    key_lower = key_name.lower()
    
    // 检查缓存
    if key_lower in vk_cache:

    // 查找VK码

    if vk_code is null:
        if len(key_lower) == 1 and key_lower.isalpha():
            vk_code = ord(key_lower.upper())
        else:
            return 0  // 未知按键
    
    // 通过DD DLL转换
    if dd_dll is available:
        dd_code = dd_dll.DD_todc(vk_code)
        if dd_code > 0:
            vk_cache[key_lower] = dd_code  // 缓存结果
            return dd_code
    
    return 0  // 转换失败
```

```
function on_delete_key(event):
    focused_widget = get_toplevel().focus_get()
```

### 6.2 快捷键绑定表

**重要：快捷键必须绑定到顶级窗口**

使用 `root.bind()` 而非 `self.bind()`，确保在画布获得焦点时快捷键仍能生效：

```
Algorithm: BindShortcuts
Input: shortcuts (List of (key, callback) pairs)

function bind_shortcuts(shortcuts):
    root = get_toplevel()
    
    for key, callback in shortcuts:
        handler = make_handler(callback, key)
        root.bind(key, handler)

function make_handler(callback, key):
    def handler(event):
        if key in ["<Delete>", "<BackSpace>"]:
            focused = get_toplevel().focus_get()
            if focused:
                widget_type = get_class_name(focused)
                if widget_type in ["CTkEntry", "Entry", "CTkTextbox", "Text"]:
                    return None  // 允许输入控件正常处理
        
        if is_callable(callback):
            callback()
        return "break"  // 阻止事件继续传播
    
    return handler
```

```
function find_tesseract():
```
