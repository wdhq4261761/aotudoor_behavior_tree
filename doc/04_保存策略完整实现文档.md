# AutoDoor 行为树系统 — 保存策略完整实现文档

## 1 概述

AutoDoor 行为树系统实现了多层次的保存策略，确保用户数据的安全性和完整性。系统包含以下七种保存机制：

| 保存类型 | 触发方式 | 存储位置 | 主要目的 |
|---------|---------|---------|---------|
| 自动保存 | 定时/变更触发 | `backup/autosave_*.json.gz` | 防止数据丢失 |
| 手动保存 | Ctrl+S / 工具栏 | 项目目录或指定路径 | 用户主动保存 |
| 另存为 | Ctrl+Shift+S | 新项目目录 | 创建新项目 |
| 项目保存 | ProjectManager | 项目根目录 | 完整项目持久化 |
| 退出保存 | 窗口关闭 | 当前项目路径 | 确保退出时数据完整 |
| 崩溃恢复 | 异常捕获 | `recovery/crash_*.json` | 异常退出后恢复 |
| 自动备份 | 项目保存时 | 项目根目录 | 版本历史管理 |

```python
def _rotate_and_save(self, data: Dict[str, Any]) -> None:
    # 备份文件轮转
    autosave_3 = self._autosave_dir / "autosave_3.json.gz"
    autosave_2 = self._autosave_dir / "autosave_2.json.gz"
    autosave_1 = self._autosave_dir / "autosave_1.json.gz"
    
    # 删除最旧的备份
    if autosave_3.exists():
        autosave_3.unlink()
    
    # 轮转备份
    if autosave_2.exists():
        autosave_2.rename(autosave_3)
    if autosave_1.exists():
        autosave_1.rename(autosave_2)
    
    # 保存新备份（gzip压缩）
    with gzip.open(autosave_1, 'wt', encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
```

```python
def save_tree(self, file_path: Optional[str] = None, save_as: bool = False):
    # 1. 如果在项目模式下且非另存为
    if self.project_root and self.project_manager and not save_as:
        tree_data = self.canvas.get_tree_data()
        
        # 1.1 收集并导入外部资源
        external_resources = ResourceService.collect_external_resources(
            tree_data, self.project_root
        )
        
        if external_resources:
            path_mapping = ResourceService.import_resources_to_project(
                external_resources, self.project_root
            )
            tree_data = ResourceService.update_tree_paths(tree_data, path_mapping)
        
        # 1.2 保存项目
        self.project_manager.save_project(tree_data)
        self._set_modified(False)
        return
    
    # 2. 如果没有项目或另存为
    if not self.project_root or save_as:
        self._on_new_project_dialog()  # 显示新建项目对话框
        return
    
    # 3. 文件保存模式
    if not file_path:
        file_path = filedialog.asksaveasfilename(...)
    
    if file_path:
        data = self.canvas.get_tree_data()
        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        self.file_path = file_path
        self._set_modified(False)
```

```python
def save_project(self, tree_data: Dict[str, Any]) -> None:
    # 1. 创建备份
    self._create_backup()
    
    # 2. 保存行为树文件
    tree_file = os.path.join(self.project_root, "tree.json")
    with open(tree_file, 'w', encoding='utf-8') as f:
        json.dump(tree_data, f, indent=2, ensure_ascii=False)
    
    # 3. 更新项目配置
    project_config = self.load_project()
    project_config["project_info"]["modified_at"] = datetime.now().isoformat()
    
    with open(os.path.join(self.project_root, "project.json"), 'w', encoding='utf-8') as f:
        json.dump(project_config, f, indent=2, ensure_ascii=False)
```

### 4.3 自动备份机制

```python
def _create_backup(self) -> None:
    tree_file = os.path.join(self.project_root, "tree.json")
    
    if not os.path.exists(tree_file):
        return
    
    # 生成备份文件名
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_file = os.path.join(
        self.project_root,
        f"tree_backup_{timestamp}.json"
    )
    
    # 复制文件
    shutil.copy2(tree_file, backup_file)
    
    # 清理旧备份，保留最近5个
    self._clean_old_backups()
```

```python
def _on_close(self):
    # 1. 检查是否有行为树编辑器
    if hasattr(self, 'behavior_tree') and self.behavior_tree:
        # 2. 记录最后打开的文件路径
        file_path = self.behavior_tree.file_path
        if file_path:
            self._settings.set_last_file_path(file_path)
        
        # 3. 检查是否有未保存的改动
        if hasattr(self.behavior_tree, '_modified') and self.behavior_tree._modified:
            result = messagebox.askyesnocancel(
                "未保存的改动",
                "当前项目有未保存的改动。\n\n是否保存？"
            )
            
            if result is None:      # 取消
                return
            elif result:            # 是 - 保存
                self.behavior_tree.save_tree()
        
        # 4. 销毁编辑器（触发自动保存）
        self.behavior_tree.destroy()
    
    # 5. 保存设置
    self._settings.save_settings()
    
    # 6. 销毁窗口
    self.destroy()
```

### 6.2 异常捕获机制

```python
def install(self) -> None:
    """安装异常处理器"""
    self._recovery_dir.mkdir(parents=True, exist_ok=True)
    self._original_excepthook = sys.excepthook
    sys.excepthook = self._handle_exception
    self._is_installed = True

def _handle_exception(self, exc_type, exc_value, exc_tb):
    """处理未捕获的异常"""
    try:
        # 保存崩溃现场数据
        self._save_crash_recovery(exc_type, exc_value, exc_tb)
        # 记录崩溃日志
        self._log_crash(exc_type, exc_value, exc_tb)
    except Exception:
        pass
    
    # 调用原始异常处理器
    if self._original_excepthook:
        self._original_excepthook(exc_type, exc_value, exc_tb)
```

### 6.3 崩溃数据保存

```python
def _save_crash_recovery(self, exc_type, exc_value, exc_tb):
    """保存崩溃恢复数据"""
    data = self._get_data_func()
    
    if not data:
        return None
    
    # 生成崩溃文件名
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    filename = f"crash_{timestamp}.json"
    filepath = self._recovery_dir / filename
    
    # 添加崩溃元数据
    data["metadata"]["save_type"] = "crash"
    data["metadata"]["crash_time"] = datetime.now().isoformat()
    data["metadata"]["crash_type"] = exc_type.__name__
    data["metadata"]["crash_message"] = str(exc_value)
    
    # 保存堆栈跟踪
    if exc_tb:
        tb_lines = traceback.format_exception(exc_type, exc_value, exc_tb)
        data["metadata"]["crash_traceback"] = "".join(tb_lines)
    
    # 写入文件
    with open(filepath, 'w', encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    
    return filepath
```

### 6.4 启动时恢复检测

```python
def _check_crash_recovery(self):
    """检查并处理崩溃恢复"""
    if not self._crash_recovery_handler.has_crash_recovery():
        return
    
    crash_info = self._crash_recovery_handler.get_latest_crash_info()
    
    result = messagebox.askyesno(
        "崩溃恢复",
        f"检测到上次未正常关闭的会话:\n"
        f"时间: {crash_info.get('crash_time', '未知')}\n"
        f"异常: {crash_info.get('crash_type', '未知')}\n\n"
        f"是否恢复该会话？"
    )
    
    if result:
        data = self._crash_recovery_handler.load_crash_file(crash_info["path"])
        if data:
            self.canvas.load_tree(data)
            self._set_modified(True)
            
            # 恢复文件路径
            metadata = data.get("metadata", {})
            saved_file_path = metadata.get("file_path")
            if saved_file_path and os.path.exists(saved_file_path):
                self.file_path = saved_file_path
    
    # 删除崩溃文件
    self._crash_recovery_handler.delete_crash_file(crash_info["path"])
```

## 7 自动备份机制

### 7.1 备份策略

| 备份类型 | 触发时机 | 存储位置 | 保留数量 |
|---------|---------|---------|---------|
| 自动保存备份 | 定时/变更 | `backup/autosave_*.json.gz` | 3个 |
| 项目备份 | 项目保存时 | 项目根目录 | 5个 |

### 7.2 备份文件命名规则

**自动保存备份**:
- `autosave_1.json.gz` - 最新
- `autosave_2.json.gz` - 次新
- `autosave_3.json.gz` - 最旧

**项目备份**:
- `tree_backup_20260411_195021.json`
- `tree_backup_20260411_194530.json`
- 格式: `tree_backup_YYYYMMDD_HHMMSS.json`

### 7.3 备份清理逻辑

```python
def _clean_old_backups(self, keep_count: int = 5) -> None:
    """清理旧备份文件"""
    backup_files = []
    
    # 收集所有备份文件
    for filename in os.listdir(self.project_root):
        if filename.startswith("tree_backup_") and filename.endswith(".json"):
            backup_files.append(os.path.join(self.project_root, filename))
    
    # 按修改时间排序（最新的在前）
    backup_files.sort(key=os.path.getmtime, reverse=True)
    
    # 删除超出保留数量的备份
    for old_backup in backup_files[keep_count:]:
        os.remove(old_backup)
```

## 10 最佳实践建议

### 10.1 用户操作建议

1. **定期手动保存**: 虽然有自动保存，但建议在重要修改后手动保存
2. **使用项目模式**: 项目模式提供更完整的备份和资源管理
3. **关注崩溃恢复提示**: 程序启动时如提示崩溃恢复，请及时处理

### 10.2 开发者注意事项

1. **保存前验证数据**: 确保序列化数据的完整性
2. **处理保存异常**: 捕获并记录保存过程中的异常
3. **线程安全**: 自动保存在后台线程执行，需注意线程安全
4. **资源清理**: 程序退出时正确停止自动保存和崩溃恢复处理器
